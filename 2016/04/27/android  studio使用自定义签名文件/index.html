android studio使用自定义签名文件

发表于 2016-04-27   |   分类于 android   |     |   阅读次数

在build.gradle中添加一段代码指定签名文件位置。

apply plugin: 'com.android.application'

android {
    compileSdkVersion 23
    buildToolsVersion "23.0.2"

    defaultConfig {
        applicationId "cn.go.lyqdh.carmap"
        minSdkVersion 15
        targetSdkVersion 23
        versionCode 1
        versionName "1.0"
    }
    buildTypes {
        release {
            minifyEnabled false
            proguardFiles getDefaultProguardFile('proguard-android.txt'), 'proguard-rules.pro'
        }
    }

    sourceSets {
        main {
            jniLibs.srcDirs = ['libs']
        }
    }
    signingConfigs {
        debug {
            storeFile file("C:\\Users\\Administrator\\Desktop\\taobao\\CarMap\\debug.keystore")
        }
    }
}

dependencies {
    compile fileTree(include: ['*.jar'], dir: 'libs')
    testCompile 'junit:junit:4.12'
    compile 'com.android.support:appcompat-v7:23.0.1'
    compile 'com.android.support:design:23.0.1'
    compile files('libs/baidumapapi_base_v3_6_1.jar')
    compile files('libs/baidumapapi_map_v3_6_1.jar')
    compile files('libs/baidumapapi_util_v3_6_1.jar')
    compile files('libs/BmobSDK_V3.4.3_0820.jar')
    compile files('libs/locSDK_6.13.jar')
    compile files('libs/baidumapapi_search_v3_6_1.jar')
    compile 'com.google.android.gms:play-services-appindexing:8.1.0'
}
